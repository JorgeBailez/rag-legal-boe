% Acrónimos y términos empleados en la memoria.

\newacronym{boe}{BOE}{Boletín Oficial del Estado}
\newacronym{pln}{PLN}{Procesamiento de Lenguaje Natural}
\newacronym{llm}{LLM}{Large Language Model (modelo grande de lenguaje)}
\newacronym{api}{API}{Application Programming Interface (interfaz de programación de aplicaciones)}
\newacronym{bm25}{BM25}{Best Matching 25 (función de ranking léxico probabilístico)}
\newacronym{ndcg}{nDCG}{normalized Discounted Cumulative Gain (ganancia acumulada descontada y normalizada)}
\newacronym{mrr}{MRR}{Mean Reciprocal Rank (rango recíproco medio)}
\newacronym{rrf}{RRF}{Reciprocal Rank Fusion (fusión por rangos recíprocos)}
\newacronym{ann}{ANN}{Approximate Nearest Neighbors (vecinos más cercanos aproximados)}
\newacronym{roc}{ROC}{Receiver Operating Characteristic}
\newacronym{auc}{AUC}{Area Under the Curve (área bajo la curva)}
\newacronym{nli}{NLI}{Natural Language Inference (inferencia en lenguaje natural)}
\newacronym{ac1}{AC1}{coeficiente de acuerdo AC1 de Gwet}
\newacronym{rgpd}{RGPD}{Reglamento General de Protección de Datos}
\newacronym{cpu}{CPU}{Central Processing Unit (unidad central de procesamiento)}
\newacronym{gpu}{GPU}{Graphics Processing Unit (unidad de procesamiento gráfico)}
\newacronym{xml}{XML}{eXtensible Markup Language (lenguaje de marcado extensible)}
\newacronym{json}{JSON}{JavaScript Object Notation}
\newacronym{csv}{CSV}{Comma-Separated Values (valores separados por comas)}
\newacronym{eli}{ELI}{European Legislation Identifier (identificador europeo de legislación)}
\newacronym{trec}{TREC}{Text REtrieval Conference}
\newacronym{mteb}{MTEB}{Massive Text Embedding Benchmark}
\newacronym{beir}{BEIR}{banco de pruebas heterogéneo de recuperación de información}
\newacronym{tfg}{TFG}{Trabajo Fin de Grado}

\newglossaryentry{embedding}{%
  name={embedding},
  description={Representación vectorial densa de un texto, construida de modo que textos con significado próximo quedan cerca en el espacio vectorial.}
}

\newglossaryentry{chunk}{%
  name={chunk (fragmento)},
  description={Unidad de texto en que se divide un documento para su indexación y recuperación.}
}

\newglossaryentry{pooling}{%
  name={pooling},
  description={Técnica de la tradición TREC para construir juicios de relevancia reuniendo los primeros resultados de varios sistemas y anotando solo ese conjunto común, de modo que la evaluación no quede sesgada a favor de un recuperador.}
}

\newglossaryentry{cuantizacion}{%
  name={cuantización},
  description={Reducción de la precisión numérica de los pesos de un modelo (por ejemplo, de 16 a 4 bits) para disminuir su consumo de memoria y de cómputo con una pérdida de calidad contenida.}
}
